cmake_minimum_required(VERSION 3.14)
project(TungstenReflect LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_BUILD_RPATH_USE_ORIGIN TRUE)

add_executable(TungstenReflect
    src/generated/projectDefines.hpp
    src/main.cpp
)

target_compile_features(TungstenReflect PUBLIC cxx_std_20)

target_link_libraries(TungstenReflect PRIVATE @PROJECT_NAME@)
