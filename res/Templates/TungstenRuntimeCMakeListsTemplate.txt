cmake_minimum_required(VERSION 3.14)
project(TungstenRuntime LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 20)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

add_executable(@EXECUTABLE_TARGET_NAME@
    src/generated/projectDefines.hpp
    src/main.cpp
)

target_link_libraries(@EXECUTABLE_TARGET_NAME@ PRIVATE @PROJECT_NAME@)

set_target_properties(@EXECUTABLE_TARGET_NAME@ PROPERTIES OUTPUT_NAME "@EXECUTABLE_NAME@")
